### -Code example-
		
**7 kyu: Vowel Count**: Return the number (count) of vowels in the given string.


```javascript
function getCount(str) {
let count = 0;
let vowel = ['a','e','i','o','u']

for ( let i = 0; i < vowel.length; i++ ) {
for ( let k = 0; k < str.length; k++ ) {
if ( vowel[i] === str[k]) count++
}
}
return count
}
```
